136 PointSegIntersection<T> Triangle<T>::CheckPointSegDist(const T& origin,const T& seg,const T& P)const
137 {
138  const auto origin_P = P - origin;
139  const T unitVecAlongSeg = seg.normalise();
140 
141  // Project point onto the seg
142  const auto Px = origin + unitVecAlongSeg*T::dotProduct(unitVecAlongSeg,origin_P);
143 
144  // Origin to projected point
145  const auto origin_Px = Px - origin;
146  auto alongSeg = T::dotProduct(unitVecAlongSeg, origin_Px);
147  static const double EPSILON= 1.0e-12;
148  if(alongSeg >= -EPSILON && alongSeg <= seg.magnitude() + EPSILON)
149  {
150  return PointSegIntersection<T>(Px, (P-Px).magnitude(), true);
151  }
152  else
153  {
154  if(alongSeg < 0.0)
155  {
156  return PointSegIntersection<T>(origin,origin_P.magnitude(),false);
157  }
158  else
159  {
160  auto otherEnd = origin + seg;
161  return PointSegIntersection<T>(otherEnd, (otherEnd - P).magnitude(),false);
162  }
163  }

167 IntersectionResult<T> Triangle<T>::CalcShortestDistanceFrom(const T& p, double maxDist)const
168 {
169  const auto projectedPointData = ProjectPointOntoShapePlane(p);
170  const auto pDist = projectedPointData.Dist;
171  if(pDist > maxDist)
172  {
173  // If the perpendicular distance from point to the triangular plane
174  // is greater than the specified threshold, we don't do any other checks
175  // The distance is set to infinity so that no other calculations are performed
176  return IntersectionResult<T>(projectedPointData.Point, std::numeric_limits<double>::max());
177  }
178 
179  const T& projectedPoint = projectedPointData.Point;
180  if(IsPointWithinShapeExtrudedAlongNormal(projectedPoint))
181  {
182  // If the point was within the extruded triangle, there is nothing else
183  // to do here and we have already found the closest point.
184  return projectedPointData;
185  }
186  else
187  {
188  // Lets check intersection with the other edges now.
189  const auto P0_P1_Data = CheckPointSegDist(m_verts[0], m_edges[0],projectedPoint);
190  const auto d0 = P0_P1_Data.IRes.Dist;
191 
192  const auto P0_P2_Data = CheckPointSegDist(m_verts[0], m_edges[1], projectedPoint);
193  const auto d1 = P0_P2_Data.IRes.Dist;
194 
195  const auto P1_P2_Data = CheckPointSegDist(m_verts[1],m_edges[2],projectedPoint);
196  const auto d2 = P1_P2_Data.IRes.Dist;
197  if(d0 < d1)
198  {
199  if(d0 < d2)
200  {
201  auto dist = sqrt(d0*d0 + pDist*pDist);
202  return IntersectionResult<T>(P0_P1_Data.IRes.Point, dist);
203  }
204  else
205  {
206  auto dist = sqrt(d2*d2 + pDist*pDist);
207  return IntersectionResult<T>(P1_P2_Data.IRes.Point, dist);
208  }
209  }
210  else
211  {
212  if(d1 < d2)
213  {
214  auto dist = sqrt(d1*d1 + pDist*pDist);
215  return IntersectionResult<T>(P0_P2_Data.IRes.Point, dist);
216  }
217  else
218  {
219  auto dist = sqrt(d2*d2 + pDist*pDist);
220  return IntersectionResult<T>(P1_P2_Data.IRes.Point, dist);

242 template<typename T>
243 BarycentricCoords Triangle<T>::CalcBarycentricCoords(const T& P)const
244 {
245  // Assume P = P0 + u*P0_P1 + v*P0_P2, goal is to calculate u and v
246  // Project this equation along P0_P1 and P0_P2 and we obtain a set
247  // of equations which can then be solved.
248  // |x| = |a, b||u|
249  // |y| = |c, d||v|
250 
251  const auto P0_P = P - m_verts[0];
252  const auto x = T::dotProduct(m_edges[0],P0_P);
253  const auto y = T::dotProduct(m_edges[1],P0_P);
254 
255  const auto a = T::dotProduct(m_edges[0],m_edges[0]);
256  const auto b = T::dotProduct(m_edges[0],m_edges[1]);
257  const auto c = T::dotProduct(m_edges[1],m_edges[0]);
258  const auto d = T::dotProduct(m_edges[1],m_edges[1]);
259 
260  const auto inverseDet = 1.0/(a*d - c*b);
261  const auto u = inverseDet * (d*x - b*y);
262  const auto v = inverseDet * (-c*x + a*y);
263  return BarycentricCoords(u,v);
264 }
